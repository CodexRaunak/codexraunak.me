Degrades gracefully: if a feed is unreachable or unparseable, that feed is skipped and the build never fails. */ -}} {{- $feeds := slice (dict "name" "Hashnode" "url" "https://codexraunak.hashnode.dev/rss.xml") (dict "name" "Medium" "url" "https://medium.com/feed/@madanraunak24") -}} {{- $posts := slice -}} {{- range $feeds -}} {{- $name := .name -}} {{- with try (resources.GetRemote .url) -}} {{- with .Err -}} {{- warnf "external-posts: could not fetch %s (%s)" $name . -}} {{- else with .Value -}} {{- $data := . | transform.Unmarshal -}} {{- with $data.channel.item -}} {{- range . -}} {{- $url := replaceRE `\?source=.*$` "" .link -}} {{- $posts = $posts | append (dict "title" (.title | plainify | htmlUnescape) "url" $url "date" (time.AsTime .pubDate) "source" $name) -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- with sort $posts "date" "desc" }}
